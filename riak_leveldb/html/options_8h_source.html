25 // DB contents are stored in a set of blocks, each of which holds a

296 // Options that control write operations
297 struct WriteOptions {
298  // If true, the write will be flushed from the operating system
299  // buffer cache (by calling WritableFile::Sync()) before the write
300  // is considered complete. If this flag is true, writes will be
301  // slower.
302  //
303  // If this flag is false, and the machine crashes, some recent
304  // writes may be lost. Note that if it is just the process that
305  // crashes (i.e., the machine does not reboot), no writes will be
306  // lost even if sync==false.
307  //
308  // In other words, a DB write with sync==false has similar
309  // crash semantics as the "write()" system call. A DB write
310  // with sync==true has similar crash semantics to a "write()"
311  // system call followed by "fsync()".
312  //
313  // Default: false
314  bool sync;
315 
316  WriteOptions()
317  : sync(false) {
318  }
319 };
