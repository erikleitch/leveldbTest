188 inline bool ParseInternalKey(const Slice& internal_key,
189  ParsedInternalKey* result) {
190  const size_t n = internal_key.size();
191  if (n < 8) return false;
192  uint64_t num = DecodeFixed64(internal_key.data() + n - 8);
193  unsigned char c = num & 0xff;
194  result->sequence = num >> 8;
195  result->type = static_cast<ValueType>(c);
196  result->user_key = Slice(internal_key.data(), n - 8);
197  return (c <= static_cast<unsigned char>(kTypeValue));
198 }
